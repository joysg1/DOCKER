++ INSTALAR DOCKER (MANJARO)
--- wget https://download.docker.com/linux/static/stable/x86_64/docker-28.4.0.tgz -qO- | tar xvfz - docker/docker --strip-components=1
--- cp -rp ./docker /usr/local/bin/ && rm -r ./docker
--- sudo pacman -U ./docker-desktop-x86_64.pkg.tar.zst
--- URL para mayor referencia: https://docs.docker.com/desktop/setup/install/linux/archlinux/

++ INSTALAR DOCKER (UBUNTU)
--- sudo snap install docker.io
--- sudo apt install docker.io -y

++ CONOCER STATUS DEL SERVICIO
--- systemctl status docker

++ HABILITAR EL INICIO DE DOCKER AL INICIAR EL EQUIPO
--- sudo systemctl enable docker

++ INICIAR EL SERVICIO
--- sudo systemctl start docker

++ DETENER EL SERVICIO
--- sudo systemctl stop docker

++ BUSCAR IMÁGENES
--- docker search 

++ DESCARGAR IMÁGEN
--- docker pull

++ LISTAR CONTENEDORES
--- sudo docker ps
--- sudo docker ps -a (lista todos los contenedores, tanto aquellos funcionando como el historial completo)

++ LISTAR IMÁGENES
--- sudo docker images

++ INICIAR CONTENEDOR
--- sudo docker run

++ REMOVER CONTENEDOR
--- sudo docker rm

++ REMOVER IMÁGEN
-- sudo docker rmi

++ EJECUTAR UN CONTENEDOR
--- sudo docker exec

++ CREAR UNA NUEVA IMAGEN DESDE UN DOCKER FILE
--- sudo docker build
